## Generando histograma...

Test 3: M√∫ltiples Subplots

import matplotlib.pyplot as plt
import numpy as np

print("Generando panel con m√∫ltiples gr√°ficos...")
## Generando panel con m√∫ltiples gr√°ficos...
fig, axes = plt.subplots(1, 2, figsize=(12, 5))

# Subplot 1
x = np.linspace(0, 10, 100)
axes[0].plot(x, np.sin(x), 'b-', label='sin(x)', linewidth=2)
axes[0].set_title('Funci√≥n Seno')
axes[0].legend()
axes[0].grid(True, alpha=0.3)

# Subplot 2
axes[1].scatter(np.random.rand(50), np.random.rand(50),
               c=np.random.rand(50), s=100, alpha=0.6, cmap='viridis')
axes[1].set_title('Puntos Aleatorios')
axes[1].grid(True, alpha=0.3)

plt.tight_layout()
plt.show()  # IMPORTANTE: necesario para RMarkdown
